DESCRIBE `user_pots`;

-- Новая таблица для управления горшками пользователей
-- Каждый пользователь может иметь до N активных горшков

id	int	NO	PRI	NULL	auto_increment
login	varchar(255)	NO		NULL		-- логин пользователя
pot_id	int	NO		NULL		-- ID горшка (1, 2, 3, 4, 5...)
pot_left	varchar(10)	YES		50%		-- позиция горшка слева (в %)
pot_top	varchar(10)	YES		80%		-- позиция горшка сверху (в %)
is_active	tinyint(1)	YES		1		-- активен ли горшок (1=да, 0=нет)
created_at	timestamp	YES		CURRENT_TIMESTAMP	-- дата создания

INDEXES:
unique_user_pot (login, pot_id) -- UNIQUE - комбинация login+pot_id уникальна
idx_login (login) -- для быстрого поиска всех горшков пользователя

ОПИСАНИЕ:
- Эта таблица хранит информацию о всех горшках пользователя
- pot_id - уникальный номер горшка для данного пользователя (1-5)
- pot_left, pot_top - позиция горшка на экране в процентах
- is_active - флаг, показывающий, активен ли горшок (можно деактивировать вместо удаления)
