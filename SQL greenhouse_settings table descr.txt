DESCRIBE `greenhouse_settings`;

-- Таблица настроек оранжереи для каждого пользователя
-- Управляет лимитами и параметрами оранжереи

id	int	NO	PRI	NULL	auto_increment
login	varchar(255)	NO	UNI	NULL		-- логин пользователя (уникальный)
max_pots	int	NO		5		-- максимальное количество горшков (по умолчанию 5)
active_pots	int	NO		1		-- текущее количество активных горшков
last_updated	timestamp	YES		CURRENT_TIMESTAMP ON UPDATE	-- дата последнего обновления

INDEXES:
login (UNIQUE) -- логин пользователя уникален
idx_login -- индекс для быстрого поиска

ОПИСАНИЕ:
- Эта таблица хранит настройки оранжереи для каждого пользователя
- max_pots - максимальное количество горшков, которое может иметь пользователь
- active_pots - текущее количество активных горшков
- В будущем можно добавить другие настройки (темы оформления, звуки и т.д.)
